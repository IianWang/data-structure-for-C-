bign add(bign A, bign B, bign C){
    //适用于正数加正数，如果是一正一负（如果正数绝对值大于负数，则直接使用减法函数），（如果负数绝对值大与正数，则直接使用减法函数后在结果前面添加负号）
    //如果是两个负数，则使用该函数后给结果前加负号
    int carry = 0;
    for (int i = 0; i < A.len || i < B.len; i++) {
        int temp = A.d[i] + B.d[i] + carry;
        C.d[C.len++] = temp % 10;
        carry = temp / 10;
    }
    if (carry != 0) {//最后一次的进位如果有，则加到最后面
        C.d[C.len++] = carry;
    }
    return C;
}

//该题是两个浮点数相加
//运用大整数思想
//把整数部分与小数部分分开存放在结构体中的两个变量
//add函数新增小数部分的相加，此处有个技巧，小数部分不同整数部分，顺序存放即可，到时候从后向前加，eg:12.11+55.321，因为数组初始化都是0，故小数部分依着最长的为准，
//也就是12.110+55.321
#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <map>
#include <math.h>
#define maxn 100100

using namespace std;

struct bign{
    int it[maxn];
    int dc[maxn];
    int len_it;
    int len_dc;
    bign(){
        memset(it, 0, sizeof(it));
        memset(dc, 0, sizeof(dc));
        len_dc = 0;
        len_it = 0;
    }
}Abign, Bbign, Cbign;

bign change(char str[]){
    int temp = 0;
    bign A;
    int i, j, k;
    for (i = 0; str[i] != '.'; i++);
    for (j = i - 1; j >= 0; j--) {//整数部分逆序存储,方便计算
        A.it[A.len_it++] = str[j] - '0';
    }
    for (k = i + 1; k < strlen(str); k++) {//小数部分顺序存储，方便计算
        A.dc[A.len_dc++] = str[k] - '0';
    }
    
    return A;
}

bign add_float(bign A, bign B){
    bign C;
    int mx = max(A.len_dc, B.len_dc);
    C.len_dc = mx;
    int carry = 0;
    for (int i = mx - 1; i >= 0; i--) {
        int temp = A.dc[i] + B.dc[i] + carry;
        C.dc[i] = temp % 10;
        carry = temp / 10;
    }
    while (C.dc[C.len_dc - 1] == 0) {//删除小数点末尾的0
        C.len_dc--;
    }
    //浮点数对整数进位carry
    int carry2 = carry;
    for (int j = 0; j < A.len_it || j < B.len_it; j++) {
        int temp = A.it[j] + B.it[j] + carry2;
        C.it[C.len_it++] = temp % 10;
        carry2 = temp / 10;
    }
    if (carry2 != 0) {
        C.it[C.len_it++] = carry2;
    }
    return C;
}

#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <map>
#include <math.h>
#define maxn 100100

using namespace std;

struct bign{
    int ls[1000];
    int len;
    bign(){
        memset(ls, 0, sizeof(ls));
        len = 0;
    }
}nodeA, nodeB;

bign serve(bign A){
    bign B;
    for (int i = A.len - 1; i >= 0; i--) {
        B.ls[B.len++] = A.ls[i];
    }
    return B;
}

bign add(bign A, bign B){
    bign C;
    int carry = 0;
    for (int i = 0; i < A.len || i < B.len; i++) {
        int temp = A.ls[i] + B.ls[i] + carry;
        C.ls[C.len++] = temp % 10;
        carry = temp / 10;
    }
    if (carry) {
        C.ls[C.len++] = carry;
    }
    return C;
}

bool judge(bign A){
    bool flag = true;
    for (int i = 0; i <= A.len / 2; i++) {
        if (A.ls[i] != A.ls[A.len - i - 1]) {
            flag = false;
            break;
        }
    }
    return flag;
}

bign trans(long long n){
    bign A;
    do{
        A.ls[A.len++] = n % 10;
        n = n / 10;
    }while(n != 0);
    return A;
}

int main(){
    long long N;
    int K;
    scanf("%lld %d", &N, &K);
    nodeA = trans(N);
    int times = 0;
    while (!judge(nodeA) && K > times) {
        nodeB = serve(nodeA);
        nodeA = add(nodeA, nodeB);
        times++;
    }
    for (int i = 0; i < nodeA.len; i++) {
        printf("%d", nodeA.ls[nodeA.len - i - 1]);
    }
    printf("\n");
    printf("%d\n", times);
        
    return 0;
};
