//问题 A: 守形数

#include <iostream>
#include <stdio.h>
#include <math.h>
#include <string.h>
#include <algorithm>

const double odds = 0.65;

#define Cpa(a, b) ((a) > (b))
#define maxn 100100

#include <map>

int digit(int n){
    int num = 0;
    while(n != 0){
        n /= 10;
        num++;
    }
    return num;
}


using namespace std;
int main()
{
    int n;
    while(scanf("%d", &n) != EOF){
        int num_digit = digit(n), N = n * n, mask = 1;
        for(int i = 0; i < num_digit; i++){
            mask *= 10;
        }
        if(N % mask == n){
            printf("Yes!\n");
        }
        else printf("No!\n");
    }

    return 0;
}

//解决办法：用进制转换的思想自定义个函数
//函数内部将整数每位单独存放，再单独重组成新的数
//注意1100 应该存放成 0011，所以避免出现bug，每次数组开始先初始化{0}

#include <stdio.h>
#include <math.h>
#include <string.h>
#include <algorithm>

const double odds = 0.65;

#define Cpa(a, b) ((a) > (b))
#define maxn 100100

#include <map>

int trans(int A[], int n){
    int index = 3;
    while(n != 0){
        A[index--] = n % 10;

        n = n / 10;
    }
    int sum = 0;
    index = 0;

    while(index != 4){
        sum += A[index] * pow(10, index);
        index++;
    }
    return sum;
}


using namespace std;
int main()
{
    int n = 1000, k;
    while(n < 10000){
        int ls[4] = {0};
        k = trans(ls, n);
        if(k == n * 9) {
            printf("%d\n", n);
            n++;
        }
        else n++;
    }

    return 0;
}
