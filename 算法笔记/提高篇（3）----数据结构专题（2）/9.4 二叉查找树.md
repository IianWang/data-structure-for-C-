//1043 Is It a Binary Search Tree (25 分)
//复习！

//首先该题首先要明确二叉树与二叉查找树的区别，二叉查找树的特定"左子叶 < 父节点 < 右子叶"，故可以在没有中序序列的情况下凭借着先序或者后序任意一条就能确定树的结构。考察点来了

//二刷，该题需要定义一个insert函数打开局面，该函数用于建立二叉查找树，二叉查找树的前序输出等于输入数据的顺序（题目有说输入数据是二叉查找树的前序输出），
//另外需要递归将左右子树交换生成镜像，借用swap
//开始还是困在了通过一个前序输出复原二叉查找树，如果说常规的树单凭一个前序输出是没有办法复原的，不过基于二叉查找树的性质定义一个insert函数，递归判断，大于等于的插右边，小于都插左边。
//不过该函数需要每输入一个数据的时候调用一次
#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <map>
#include <math.h>
#include <vector>
#include <set>
#include <queue>
#include <stack>
#define maxn 110
using namespace std;
int N;
vector<int> serise;

bool flag = true;

struct node{
    int data;
    node* left;
    node* right;
    node(){
        left = NULL;
        right = NULL;
    }
}ls[1001];

node* root = NULL;

void insert(node* &root, int val){
    if (root == NULL) {
        root = new node; //这里不要再定义一次 node* root，因为出入进来的root已经是node*类型的，否则就是两个root，主程序的那个没有参与
        root -> data = val;
        return;
    }
    if (root -> data > val) insert(root -> left, val);
    else insert(root -> right, val);
}


void preorder(node* root){
    if (root == NULL) {
        return;
    }
    serise.push_back(root -> data);
    preorder(root -> left);
    preorder(root -> right);
}

void postorder(node* root){
    if (root == NULL) {
        return;
    }
    postorder(root -> left);
    postorder(root -> right);
    serise.push_back(root -> data);
//    printf("%d", root -> data);
}

void mirror(node* root){
    if (root == NULL) {
        return;
    }
    else{
        node* temp = root -> left;
        root -> left = root -> right;
        root -> right = temp;
        mirror(root -> left);
        mirror(root -> right);
    }
}

void print(){
    flag = false;
    printf("YES\n");
    serise.clear();
    postorder(root);
    for (int i = 0; i < serise.size(); i++) {
        printf("%d", serise[i]);
        if (serise.size() - i != 1) {
            printf(" ");
        }
    }
    printf("\n");
}

int main(){
    vector<int> vt;
    scanf("%d", &N);
    
    for (int i = 0; i < N; i++) {
        int temp;
        scanf("%d", &temp);
        vt.push_back(temp);
        insert(root, temp);
    }
    
    preorder(root);
    if (serise == vt) print();
    else{
        mirror(root);
        serise.clear();
        preorder(root);
        if (serise == vt) print();
    }
    if (flag) printf("NO\n");

    
    return 0;
};

//1064 Complete Binary Search Tree (30 分)
//复习完全二叉树的存储特性

//以中序遍历输入的序列(一定要确保输入序列是从小到大的，先排个序)，一边遍历一边存储到一个新的序列中。
//遍历之后该序列的输出自然是层序输出。

//特性：中序遍历对从小到大的序列进行读取得到的序列就是完全二叉树的层序输出

#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <map>
#include <math.h>
#include <vector>
#include <set>
#include <queue>
#include <stack>
#define maxn 1010
using namespace std;
int N, ls[maxn], CBT[maxn], ind = 0;

void inOrder(int root){
    if (root > N) {
        return;
    }
    inOrder(root * 2);
    CBT[root] = ls[ind++];
    inOrder(root * 2 + 1);
}


int main(){
    scanf("%d", &N);
    for (int i = 0; i < N; i++) {
        scanf("%d", &ls[i]);
    }
    sort(ls, ls + N);
    inOrder(1);
    for (int i = 1; i <= N; i++) {
        printf("%d", CBT[i]);
        if (i != N) {
            printf(" ");
        }
    }
    printf("\n");
    

    
    return 0;
};
