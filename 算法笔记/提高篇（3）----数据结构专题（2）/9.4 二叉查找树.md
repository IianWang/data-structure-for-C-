### 9.4.3 二叉查找树的性质

**关于完全二叉树，是每层从左至右填满，并不是一定每个结点都是二叉树(不算叶子结点)，如果三个叶子结点分配到两个父结点，那么就先拿两个填满左父结点，剩下一个填到右父结点**

左子叶 < 父节点 < 右子叶

```C++
//1043 Is It a Binary Search Tree (25 分)
//复习！
//该题考察点比较多。判断一段输入序列(题中有说是按先序输入)，让判断输入的这串序列是否是二叉查找树或者是镜像的二叉查找树
//首先该题首先要明确二叉树与二叉查找树的区别，二叉查找树的特定"左子叶 < 父节点 < 右子叶"，故可以在没有中序序列的情况下凭借着先序或者后序任意一条就能确定树的结构。考察点来了
//1. 根据一条先序序列建立一个二叉查找树（自定义insert函数，关键！）
//2. 将一棵树左右互换生成镜像树（自定义mirror函数）
//3. 将一棵树进行后序遍历输出（正常的后序遍历，通用的那种）

//此外：因为题目输入数据是以先序形式进行的输出，还需要新增一个先序遍历函数，以备将镜像树以先序的形式输出到vector中与输入序列对比。
//PS: vector在此题中非常好用，两个vertor直接可以进行比较，不用迭代，方便的多
#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <map>
#include <math.h>
#include <vector>
#include <set>
#include <queue>
#include <stack>
#define maxn 110
using namespace std;
int N;
vector<int> serise;

bool flag = true;

struct node{
    int data;
    node* left;
    node* right;
    node(){
        left = NULL;
        right = NULL;
    }
}ls[1001];

node* root = NULL;

void insert(node* &root, int val){
    if (root == NULL) {
        root = new node; //这里不要再定义一次 node* root，因为出入进来的root已经是node*类型的，否则就是两个root，主程序的那个没有参与
        root -> data = val;
        return;
    }
    if (root -> data > val) insert(root -> left, val);
    else insert(root -> right, val);
}


void preorder(node* root){
    if (root == NULL) {
        return;
    }
    serise.push_back(root -> data);
    preorder(root -> left);
    preorder(root -> right);
}

void postorder(node* root){
    if (root == NULL) {
        return;
    }
    postorder(root -> left);
    postorder(root -> right);
    serise.push_back(root -> data);
//    printf("%d", root -> data);
}

void mirror(node* root){
    if (root == NULL) {
        return;
    }
    else{
        node* temp = root -> left;
        root -> left = root -> right;
        root -> right = temp;
        mirror(root -> left);
        mirror(root -> right);
    }
}

void print(){
    flag = false;
    printf("YES\n");
    serise.clear();
    postorder(root);
    for (int i = 0; i < serise.size(); i++) {
        printf("%d", serise[i]);
        if (serise.size() - i != 1) {
            printf(" ");
        }
    }
    printf("\n");
}

int main(){
    vector<int> vt;
    scanf("%d", &N);
    
    for (int i = 0; i < N; i++) {
        int temp;
        scanf("%d", &temp);
        vt.push_back(temp);
        insert(root, temp);
    }
    
    preorder(root);
    if (serise == vt) print();
    else{
        mirror(root);
        serise.clear();
        preorder(root);
        if (serise == vt) print();
    }
    if (flag) printf("NO\n");

    
    return 0;
};
```

```C++
#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <map>
#include <math.h>
#include <vector>
#include <set>
#include <queue>
#include <stack>
#define maxn 1010
using namespace std;
int N, ls[maxn], CBT[maxn], ind = 0;

void inOrder(int root){
    if (root > N) {
        return;
    }
    inOrder(root * 2);
    CBT[root] = ls[ind++];
    inOrder(root * 2 + 1);
}


int main(){
    scanf("%d", &N);
    for (int i = 0; i < N; i++) {
        scanf("%d", &ls[i]);
    }
    sort(ls, ls + N);
    inOrder(1);
    for (int i = 1; i <= N; i++) {
        printf("%d", CBT[i]);
        if (i != N) {
            printf(" ");
        }
    }
    printf("\n");
    

    
    return 0;
};
```
