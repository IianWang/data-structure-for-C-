struct date{
    int year;
    int month;
    int day;
}st_dt, en_dt;

void incres(date &dt){
    if(dt.month == 4 || dt.month == 6 || dt.month == 9 || dt.month == 11){
        dt.day += 1;
        if(dt.day > 30){
            dt.day = 1;
            dt.month += 1;
            if(dt.month > 12){
                dt.year += 1;
                dt.month = 1;
            }
        }
    }
    else if(dt.month == 2){
        dt.day += 1;
        if(dt.year % 4 == 0){
            if(dt.day > 29){
            dt.day = 1;
            dt.month += 1;
            if(dt.month > 12){
                dt.year += 1;
                dt.month = 1;
            }
        }
        }
        else{
            if(dt.day > 28){
            dt.day = 1;
            dt.month += 1;
            if(dt.month > 12){
                dt.year += 1;
                dt.month = 1;
            }
        }
        }
    }
    else{
        dt.day += 1;
        if(dt.day > 31){
            dt.day = 1;
            dt.month += 1;
            if(dt.month > 12){
                dt.year += 1;
                dt.month = 1;
            }
        }
    }

}

using namespace std;
int main()
{
    int N;
    char st[9], ed[9];

    while(scanf("%d", &N) != EOF){
        int n = 0, k = 1;
        scanf("%s", st);

        sscanf(ed, "%4d%2d%2d", &en_dt.year, &en_dt.month, &en_dt.day);
        while(k){
            incres(st_dt);
            n++;
            if(st_dt.year == en_dt.year && st_dt.month == en_dt.month && st_dt.day == en_dt.day){
                k = 0;
            }
        }
        printf("%d\n", n + 1);
    }
    return 0;
}
