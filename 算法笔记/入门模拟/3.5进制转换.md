```C++
//1019 General Palindromic Number (20分)
//输出进制转化结果后还需判断是否是回文（Palindromic）数字，也就是数字是否关于中间那个数字对称
//bool函数判断回文数字，然后if调用bool函数
#include <iostream>
#include <stdio.h>
#include <math.h>
#include <string.h>
using namespace std;

bool judge(int ls[], int n){
    for (int i = 0; i <n; i++) {
        if(ls[i] != ls[n - i - 1]) return false;
    }
    return true;
}

int main(){
    int N, b, ls[100], n = 0;
    scanf("%d %d", &N, &b);
    do{
        ls[n] = N % b;
        N = N / b;
        n++;
    }while (N != 0);
    
    if (judge(ls, n)) {
        printf("Yes\n");
    }
    else printf("No\n");
    for (int i = 0; i <n; i++) {
        printf("%d", ls[n - i - 1]);
        if(n - i != 1) printf(" ");
    }
    printf("\n");
    
    return 0;
}
```

```C++
//1058 A+B in Hogwarts (20分)
//货币换算
//看似简单，实则也简单的一题，注意用余数和除数来做，打个比方（"角"满十进"一元"）
//但是数据点可能给出11.22.1(11块22角1分 = 13块2角1分)
//故，进位用整除后的商(22 / 10 = 2)，余数作为进位后的不满位的结果(22 % 10 = 2)
//写法一，推荐写法二
#include <iostream>
#include <stdio.h>
#include <math.h>
#include <string.h>
using namespace std;

struct cur{
    int K;
    int S;
    int G;
}cur_A, cur_B, temp;

void add(cur A, cur B, cur &C){
    if(A.K + B.K >= 29){
        C.K = (A.K + B.K) % 29;
        A.S += (A.K + B.K) / 29;
    }
    else C.K = A.K + B.K;
    
    if (A.S + B.S >= 17) {
        C.S = (A.S + B.S) % 17;
        A.G += (A.S + B.S) / 17;
    }
    else C.S = A.S + B.S;
    
    C.G = A.G + B.G;
}

int main(){
    scanf("%d.%d.%d", &cur_A.G, &cur_A.S, &cur_A.K);
    scanf("%d.%d.%d", &cur_B.G, &cur_B.S, &cur_B.K);
    add(cur_A, cur_B, temp);
    printf("%d.%d.%d", temp.G, temp.S, temp.K);
    
    return 0;
}
```
