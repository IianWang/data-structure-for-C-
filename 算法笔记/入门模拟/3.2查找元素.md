// 在每次给定的区间范围内找最大，对每次找到的最大进行累乘，最后输出每次最大的位置和累乘结果。
//1011 World Cup Betting (20分)
#include <iostream>
#include <stdio.h>
#include <math.h>
#include <string.h>
#include <time.h>
const double eps = 1e-8;
const double odds = 0.65;
#define Equ(a, b) (fabs((a) - (b))<(eps))
#define Cpa(a, b) ((a) > (b))
#define maxn 1111

void fid_max(double ls[], double &odd, int &n, int k){
    odd = ls[k];
    n = k;
    for(int i = k; i <k + 3; i++){
        if(ls[i] > odd){
            odd = ls[i];
            n = i;
        }
    }
}

using namespace std;
int main()
{
    double ls[10], odd = 1.0, total = 1.0;
    int n = 0;
    char ls_seq[4];
    char ls_str[4] = "WTL";
    int k = 0;
    for(int i = 0; i <9; i++){
        scanf("%lf", &ls[i]);
    }

    for(int i = 0; i <3; i++){
        fid_max(ls, odd, n, k);
        total *= odd;
        ls_seq[n / 3] = ls_str[n % 3];
        ls[n] = 0.0;
        odd = 1.0;
        n = 0;
        k += 3;
    }

    for(int i = 0; i <3; i++){
        printf("%c ", ls_seq[i]);
    }
    printf("%.2lf", (total * odds - 1) * 2);
    return 0;
}
