//问题 C: 查找
//基本二分查找判断待查元素是否存在于列表中。
//注意一点，调用search函数前，先把ls排好序（最好是升序）；
#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <map>
#include <math.h>

using namespace std;
int binarySearch(int ls[], int left, int right, int aim){
    int mid;
    while (left <= right) {
        mid = (left + right) / 2;
        if (ls[mid] == aim) return mid;
        else if (ls[mid] > aim){
            right = mid - 1;
        }
        else{
            left = mid + 1;
        }
    }
    return -1;
}

int main(){
    int ls[110], ele[110], N, M;
    while (scanf("%d", &N) != EOF) {
        for (int i = 0; i < N; i++) {
            scanf("%d", &ls[i]);
        }
        sort(ls, ls + N);
        scanf("%d", &M);
        for (int i = 0; i < M; i++) {
            int temp, loc;
            scanf("%d", &temp);
            loc = binarySearch(ls, 0, N - 1, temp);
            printf("%s\n", loc == -1? "NO":"YES");
        }
    }
    
    return 0;
};
