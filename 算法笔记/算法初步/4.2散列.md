#include <math.h>
#include <string.h>
#include <algorithm>

const double odds = 0.65;

#define Cpa(a, b) ((a) > (b))

int a[110] = {0};

{

    while(m--){

        memset(a, 0, sizeof(a));

        scanf("%d", &n);
        int max_ = 0;
        for(int i = 1; i <=n; i++){
            scanf("%d", &a[i]);
            if(a[i] > max_) max_ = a[i];
            hashTable[0][a[i]] = 1;
        }
        int max_group = 0;
        for(int i = 1; i <=n; i++){
            int x;
            scanf("%d", &x);
            hashTable[x][a[i]]++;
            if(x > max_group) max_group = x;
            hashTable[x][0] = 1;
        }
        for(int i = 1; i <=max_group; i++){
            if(hashTable[i][0] == 0) continue;
            printf("%d={", i);
            for(int j = 1; j <=max_; j++){
                if(hashTable[0][j] == 0) continue;
                printf("%d=%d", j, hashTable[i][j]);
                if(j != max_) printf(",");
            }
            printf("}", i);
            printf("\n");
        }

    }




    return 0;
}
