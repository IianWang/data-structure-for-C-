//1093 Count PAT's (25 分)
//这道题看似通过率蛮高的，不过就我第一次做而言，我只想到了暴力for(for)，最大长度为10^5，这显然是超时的
//决胜技巧在与PAT三个字母的个数计算，以A为中点，统计它左边P的个数再统计它右边T的个数，把所有的A都统计一遍取和就是最终的值
//！！第二个重点在于“Output Specification”提到的
//“Since the result may be a huge number, you only have to output the result moded by 1000000007.”
//要求将结果除1000000007取余数输出
//如果没有读到这个轻则多花些时间才找到这块，重则花了时间也找不到bug，两个数据点9分！

#include <iostream>
#include <stdio.h>
#include <math.h>
#include <string.h>
#include <algorithm>
const double eps = 1e-8;
const double odds = 0.65;
#define Equ(a, b) (fabs((a) - (b))<(eps))
#define Cpa(a, b) ((a) > (b))
#define maxn 100100
#include <map>
char str[maxn];
struct node{
    long long countP;
    long long countT;
    int locA;
}ls[maxn];


using namespace std;
int main()
{
    scanf("%s", str);
    int len = strlen(str);

    long long amountP = 0, amountT = 0;
    for(int i = 0; i < len; i++){
        if(str[i] == 'P') amountP++;
        if(str[i] == 'A') {
            ls[i].countP = amountP;
            ls[i].locA = 1;
        }
    }
    long long amount = 0;
    for(int i = len - 1; i >= 0; i--){
        if(str[i] == 'T') amountT++;
        if(str[i] == 'A') {
            amount += ls[i].countP * amountT;
        }
    }

    printf("%lld\n", amount % 1000000007);

    return 0;
}
