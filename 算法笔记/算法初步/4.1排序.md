#include <iostream>
#include <stdio.h>
#include <math.h>
#include <string.h>

using namespace std;
int main()
{
    int ls[100], N;
    while(scanf("%d", &N) != EOF){
        int n = 0;
        while(N--){
            scanf("%d", &ls[n++]);
        }

        int k;
        for(int i = 0; i <n; i++){
            k = i;
            for(int j = i; j <n; j++){
                if(ls[k] > ls[j]) k = j;
            }
            int temp;
            temp = ls[k];
            ls[k] = ls[i];
            ls[i] = temp;
        }

        for(int i = 0; i <n; i++){
            printf("%d", ls[i]);
            if(n - i != 1) printf(" ");
        }
        printf("\n");
    }
    return 0;
}

//问题 I: 排名
#include <iostream>
#include <stdio.h>
#include <math.h>
#include <string.h>
#include <algorithm>
const double eps = 1e-8;
const double odds = 0.65;
#define Equ(a, b) (fabs((a) - (b))<(eps))
#define Cpa(a, b) ((a) > (b))
#define maxn 1111

struct stu{
    char no[21];
    int pass;
    int tot;
}ls[1000];

bool cmp(stu a, stu b){
    if(a.tot != b.tot) return a.tot > b.tot;
    else return strcmp(a.no, b.no) < 0;
}

using namespace std;
int main()
{
    int N, M, G;

    while(scanf("%d", &N) != EOF){
        if(N == 0) break;
        scanf("%d %d", &M, &G);

        int score[10];
        for(int i = 0; i <M; i++){
            scanf("%d", &score[i]);
        }
        for(int i = 0; i <N; i++){
            scanf("%s %d", ls[i].no, &ls[i].pass);
            int temp;
            for(int j = 0; j <ls[i].pass; j++){
                scanf("%d", &temp);
                if(j == 0) ls[i].tot = 0;
                ls[i].tot += score[temp - 1];
            }
        }

        sort(ls, ls + N, cmp);

        int n = 0;
        for(int i = 0; i <N; i ++){
            if(ls[i].tot >= G) n++;
        }
        printf("%d\n", n);

        for(int i = 0; i <N; i ++){
            if(ls[i].tot >= G) printf("%s %d\n", ls[i].no, ls[i].tot);
        }
    }




    return 0;
}
