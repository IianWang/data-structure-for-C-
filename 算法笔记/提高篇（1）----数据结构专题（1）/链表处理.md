void delet_ele(node *head, int x){
    //对链表来说删除元素是指删掉链表上所有的与给定值相等的元素
    node* p = head -> next;
    node* pre = head;
    while (p) {
        if (p -> data == x) {
            pre -> next = p -> next;
            delete(p);
            p = pre -> next;
        }
        else{
            pre = p;
            p = p -> next;
        }
    }
}
