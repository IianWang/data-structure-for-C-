### 7.3.3 链表的基本操作

- 查找元素
```C++
void insert_ele(node *head, int pos, int x){
    node* p = head;//直接从首元结点开始循环
    for (int i = 0; i < pos - 1; i++) {
        p = p -> next;
    }
    node* now = new node;
    now -> data = x;
    now -> next = p -> next;
    p -> next = now;
}
```
- 插入元素
```C++
void insert_ele(node *head, int pos, int x){
    node* p = head;//直接从首元结点开始循环
    for (int i = 0; i < pos - 1; i++) {
        p = p -> next;
    }
    node* now = new node;
    now -> data = x;
    now -> next = p -> next;
    p -> next = now;
}
```
- 删除元素
```C++
void delet_ele(node *head, int x){
    //对链表来说删除元素是指删掉链表上所有的与给定值相等的元素
    node* p = head -> next;
    node* pre = head;
    while (p) {
        if (p -> data == x) {
            pre -> next = p -> next;
            delete(p);
            p = pre -> next;
        }
        else{
            pre = p;
            p = p -> next;
        }
    }
}
```
