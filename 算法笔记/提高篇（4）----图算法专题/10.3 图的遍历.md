//该题采用DFS对单个路口进行遍历，travlDFS即为遍历所有路口。
//统计每个路口沿途经过的点数、总权重、最重的点。所以DFS内部要能实现这三者，此处必须一提的给DFS增加引用参数统计总权重最重的点、经过的点数。
//输入数据都为字符串，故需要给字符串转化为数字，理念是这个操作越简单越好，不要搞的太复杂，故赋一个初值pointer = 0，出现一个新字符串用pointer代表它，然后pointer++为第二个新字符准备
//这也就表明需要map存储string-int的映射

#include <iostream>
#include<bits/stdc++.h>
const double eps = 1e-8;
const double odds = 0.65;
#define Equ(a, b) (fabs((a) - (b))<(eps))
#define Cpa(a, b) ((a) > (b))
#define maxn 10051
const int INF = 1000000000;
using namespace std;
int ls[maxn][maxn] = {0}, pointer = 0;
map<string, int> strToint;
map<int, string> intTostr;
map<string, int> mp;
int ls_weight[maxn] = {0}, N, K;
bool hashls[maxn] = {false};

int converse(string str){
    if(strToint.find(str) == strToint.end()){
        strToint[str] = pointer;
        intTostr[pointer] = str;
        pointer++;
        return pointer - 1;
    }
    return strToint[str];

}

void DFS(int root, int &head, int &member, int &tot_weight){
    member++;
    hashls[root] = true;
    if(ls_weight[root] > ls_weight[head]) head = root;
    for(int i = 0; i < pointer; i++){
        if(ls[root][i] > 0){
            tot_weight += ls[root][i];
            ls[root][i] = 0;
            if(hashls[i] == false){
                DFS(i, head, member, tot_weight);
            }
        }
    }
}

void travlDFS(){
    for(int i = 0; i < N; i++){
        if(hashls[i] == false){
            int member = 0, head = i, tot_weight = 0;
            DFS(i, head, member, tot_weight);
            if(member > 2 && tot_weight > K){
                mp[intTostr[head]] = member;
            }
        }
    }
}

int main(){
    scanf("%d %d", &N, &K);
    for(int i = 0; i < N; i++){
        int weight;
        string a, b;
        cin >> a >> b >> weight;
        int x, y;
        x = converse(a);
        y = converse(b);
        ls[x][y] = weight;
        ls_weight[x] += weight;
        ls_weight[y] += weight;
    }
    travlDFS();
    int gp = 0;

            printf("%s %d\n", (it -> first).c_str(), it -> second);
        }
    }

    return 0;
}

//1076 Forwards on Weibo (30 分)
//数组矩阵 + BFS解决问题，但是有几个点需要特别注意。
//1. 图的BFS虽然遵循树BFS常用的固定模式，但多多少少还有多点点不同，就是图的遍历（特别是数组矩阵），多了一个bool hash[]数组，

#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <map>
#include <math.h>
#include <vector>
#include <set>
#include <queue>
#include <stack>
#define maxn 2010
using namespace std;
int ls[maxn][maxn] = {0}, N, levl;
bool vis[maxn] = {false};
struct node{
    int user_num;
    int level;
};


int BFS(node first){
    queue<node> qu;
    int amount = 0;
    vis[first.user_num] = true;
    qu.push(first);
    while (!qu.empty()) {
        int level_new = 0;
        bool flag = true;
        node temp = qu.front();
        qu.pop();
        for (int i = 1; i <= N; i++) {
            if (ls[i][temp.user_num] == 1 && vis[i] == false) {
                if (flag) {
                    flag = false;
                    level_new = temp.level + 1;
                }
                amount++;
                vis[i] = true;
                node new_temp;
                new_temp.user_num = i;
                new_temp.level = level_new;
                qu.push(new_temp);
            }
        }
        if (qu.front().level == levl) {
            break;
        }
    }
    return amount;
}

int main(){
    scanf("%d %d", &N, &levl);
    int counts, flw;

        scanf("%d", &counts);

//思路：这种题信息复杂，要先停下来分析，如何找到令树最高的结点？（肯定至少有两个，一首一尾），就先找到层数最深的点。然后再以这个点作为根结点出发，找能到达的满足最大的所有叶子结点。

int N, M = -1;
map<int, vector<int> > ls;
vector<int> V;
set<int> S;
bool vis[maxn] = {false};

void DFS(int root, int step){
    vis[root] = true;
    if (step > M) {
        M = step;
        V.clear();
        V.push_back(root);
    }
    if (step == M) {
        //如果上个if除法了这个也会触发，第一次查找最深的root, V内一定有两个一样的
        V.push_back(root);
    }

    for (int i = 0; i < ls[root].size(); i++) {
        int new_root = ls[root][i];
        if (vis[new_root] == false) {//这里要加false，因为是双向存储，1存了2，2也存了1，遍历到2不能再跑回1.
            DFS(new_root, step + 1);
        }
    }
}

int main(){
    scanf("%d", &N);
    for (int i = 1; i < N; i++) {
        int a, b;
        scanf("%d %d", &a, &b);
        ls[a].push_back(b);
        ls[b].push_back(a);

    int counts = 0, temp = 0;
    for (int i = 1; i <= N; i++) {

        }
    }
    else printf("Error: %d components\n", counts);


    return 0;
};
