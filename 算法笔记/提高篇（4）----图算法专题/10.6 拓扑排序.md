```C++
#include <iostream>
#include<bits/stdc++.h>
const double eps = 1e-8;
const double odds = 0.65;
#define Equ(a, b) (fabs((a) - (b))<(eps))
#define Cpa(a, b) ((a) > (b))
#define maxn 1011
const int INF = 100000000;
using namespace std;
int N, M;
vector<int> ls[maxn];
int inDegree[maxn] = {0};


bool topo(){
    int num = 0;
    queue<int> qu;
    for(int i = 1; i <= N; i++){
        if(inDegree[i] == 0) qu.push(i);
    }

    while(!qu.empty()){
        int temp = qu.front();
        qu.pop();
        for(int j = 0; j < ls[temp].size(); j++){
            int v = ls[temp][j];
            inDegree[v]--;
            if(inDegree[v] == 0){
                qu.push(v);
            }
        }
        num++;//如果存在环，这个队列结束后num是要比N小的，整个环上的元素都没能入队
    }
    if(num == N) return true;
    else return false;
}


int main()
{
    scanf("%d %d", &N, &M);
    for(int i = 0; i < M; i++){
        int a, b;
        scanf("%d %d", &a, &b);
        ls[a].push_back(b);
        inDegree[b]++;


    }
    printf("%s\n", topo() ? "TRUE":"FALSE");

}
```
