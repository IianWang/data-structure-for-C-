## 8.1深度优先搜索（DFS）
Depth First Search


```C++
//1103 Integer Factorization (30 分)
//index < 1改成 index < 0结果会出现 amount > K 拦不住的情况。还没搞清楚为什么

#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <map>
#include <math.h>
#include <vector>
#include <set>
#include <queue>
#include <stack>
#define maxn 10100

using namespace std;

int N, K, P, ind = 0, max_sum = -1, test;
vector<int> temp, ans, fac;
int len = 0, ls[maxn];

void init(){
    int i = 0, temp = 0;
    while (temp <= N) {
        temp = (int)pow(i * 1.0, P * 1.0);
        fac.push_back(temp);
        i++;
    }
}

void DFS(int index, int amount, int pow_sum, int add_sum){
    if (pow_sum == N && amount == K) {
        if (add_sum > max_sum) {
            ans = temp;
            max_sum = add_sum;
            test = amount;
        }
        return;
    }
    //index < 0会出现amount出现异常
    if (index < 1 || amount > K || pow_sum > N) return;
    temp.push_back(index);
    DFS(index, amount + 1, pow_sum + fac[index], add_sum + index);
    temp.pop_back();
    DFS(index - 1, amount, pow_sum, add_sum);
}


int main(){
    scanf("%d %d %d", &N, &K, &P);
    init();
//    printf("size = %d\n", (int)fac.size());
    DFS((int)fac.size() - 1, 0, 0, 0);
    if (max_sum == -1) {
        printf("Impossible\n");
    }
    else{
        printf("%d = %d^%d", N, ans[0], P);
        for (int i = 1; i < ans.size(); i++) {
            printf(" + %d^%d", ans[i], P);
        }
    }
    printf("test = %d\n", test);

    
    
    return 0;
};
```
